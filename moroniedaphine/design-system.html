CSS Tokens

/* -- COLOR TOKENS -- */
:root {
    /* HSL base (Avela White pattern) */
  --burgundy-hsl:  350, 67%, 14%;
  --gold-hsl:      43, 59%, 52%;

    /* Hex tokens */
  --burgundy:       #3D0C10;
  --burgundy-deep:  #2B070A;
  --burgundy-mid:   #5C1A1F;
  --burgundy-tint:  #F7F2F2;
  --parchment:      #FAF8F6;  /* warm near-white */
  --gold:           #D4AF37;
  --gold-border:    rgba(212,175,55,0.30);

    /* TYPOGRAPHY */
  --font-display:   'Fraunces', Georgia, serif;
  --font-ui:        'Inter', system-ui, sans-serif;

    /* SPACING */
  --sp-1:4px;  --sp-2:8px;  --sp-3:16px;  --sp-4:24px;
  --sp-5:32px; --sp-6:48px; --sp-7:64px; --sp-8:96px;

    /* EASING */

  --ease-out:      cubic-bezier(0.16, 1, 0.3, 1);
  --ease-dramatic: cubic-bezier(0.22, 1, 0.36, 1);

    /* SHADOWS (Bliss & Bone system) */

}
